Alphabet a b c d e f g h i k l m n o p q r s t u v w x y z ;

Sets
    V = a e i o u ;
    C = b c d f g h k l m n p q r s t v w x y z ;

Rules
"-a drop before o"
a:0 <=> _ %-:0 o ;
        _ %-:0 e ;

!"-a drop before -e"
!a:0 <=> _ %-:0 e ;

"-e theme vowel in second singular passive future (1st & 2nd)"
%-:e <=> C _ r ;

"-i theme vowel between consonant s, t and m"
%-:i <=> C _ [ s | t | m ] ;

"-u theme vowel between consonant and n"
%-:u <=> C _ n ;
